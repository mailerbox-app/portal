        const SCRIPT_URL = 'https://script.google.com/macros/s/AKfycbyF71tpaUx7gyQtArVRhUCmd39TqrRxhDbo3x4KSLq1t4T3wI2A9ISi4jaqyILEK3oIVQ/exec';

        // --- NAVEGACIÓN Y VISTAS ---
        function showView(viewId) {
            // Ocultar todas
            document.getElementById('login-view').classList.add('hidden');
            document.getElementById('register-view').classList.add('hidden');
            document.getElementById('dashboard-view').classList.add('hidden');
            
            // Ocultar alertas
            document.getElementById('login-error').classList.add('hidden');
            document.getElementById('register-error').classList.add('hidden');
            document.getElementById('login-success').classList.add('hidden');

            // Actualizar UI Header
            const subtitle = document.getElementById('header-subtitle');
            const logoutBtn = document.getElementById('logout-btn');

            if (viewId === 'dashboard-view') {
                subtitle.innerText = "Detalles de tu Cuenta de Correo Comercial";
                logoutBtn.classList.remove('hidden');
            } else {
                subtitle.innerText = "Bóveda de Cliente Segura";
                logoutBtn.classList.add('hidden');
                
                // Limpiar formularios al salir
                document.getElementById('login-form').reset();
                document.getElementById('register-form').reset();
            }

            // Mostrar la seleccionada
            document.getElementById(viewId).classList.remove('hidden');
        }

        function toggleLoading(show) {
            const overlay = document.getElementById('loading-overlay');
            if (show) {
                overlay.classList.remove('hidden');
                overlay.classList.add('flex');
            } else {
                overlay.classList.add('hidden');
                overlay.classList.remove('flex');
            }
        }

        // --- LÓGICA DE API EXTERNA (FETCH) ---
        
        // Manejar Registro
        async function handleRegister(e) {
            e.preventDefault();
            const email = document.getElementById('reg-email').value;
            const pass = document.getElementById('reg-pass').value;
            const errorDiv = document.getElementById('register-error');
            
            toggleLoading(true);

            if (SCRIPT_URL === 'YOUR_GOOGLE_SCRIPT_WEB_APP_URL') {
                toggleLoading(false);
                errorDiv.innerText = "Por favor, configura tu SCRIPT_URL en el código HTML primero.";
                errorDiv.classList.remove('hidden');
                return;
            }

            try {
                // Se envía la petición POST al backend
                const response = await fetch(SCRIPT_URL, {
                    method: 'POST',
                    body: JSON.stringify({ action: 'register', email: email, password: pass }),
                    headers: {
                        'Content-Type': 'text/plain;charset=utf-8',
                    }
                });
                
                const result = await response.json();
                toggleLoading(false);

                if(result.success) {
                    showView('login-view');
                    const successDiv = document.getElementById('login-success');
                    successDiv.innerText = result.message;
                    successDiv.classList.remove('hidden');
                } else {
                    errorDiv.innerText = result.message;
                    errorDiv.classList.remove('hidden');
                }
            } catch (error) {
                toggleLoading(false);
                errorDiv.innerText = "Error del servidor: Ocurrió un problema de conexión CORS o red.";
                errorDiv.classList.remove('hidden');
                console.error(error);
            }
        }

        // Manejar Login
        async function handleLogin(e) {
            e.preventDefault();
            const email = document.getElementById('login-email').value;
            const pass = document.getElementById('login-pass').value;
            const errorDiv = document.getElementById('login-error');
            
            toggleLoading(true);

            if (SCRIPT_URL === 'YOUR_GOOGLE_SCRIPT_WEB_APP_URL') {
                toggleLoading(false);
                errorDiv.innerText = "Por favor, configura tu SCRIPT_URL en el código HTML primero.";
                errorDiv.classList.remove('hidden');
                return;
            }

            try {
                // Se envía la petición POST al backend
                const response = await fetch(SCRIPT_URL, {
                    method: 'POST',
                    body: JSON.stringify({ action: 'login', email: email, password: pass }),
                    headers: {
                        'Content-Type': 'text/plain;charset=utf-8',
                    }
                });
                
                const result = await response.json();
                toggleLoading(false);

                if(result.success) {
                    // Inyectar datos en el dashboard
                    document.getElementById('data-dominio').innerText = result.data.dominio;
                    document.getElementById('data-renovacion').innerText = result.data.renovacion;
                    document.getElementById('data-correoComercial').innerText = result.data.correoComercial;
                    document.getElementById('data-claveAcceso').innerText = result.data.claveAcceso;
                    
                    // Llenar datos de la nueva tarjeta Quick Setup
                    document.getElementById('quick-email').innerText = result.data.correoComercial;
                    document.getElementById('quick-pass').innerText = result.data.claveAcceso;
                    
                    showView('dashboard-view');
                } else {
                    errorDiv.innerText = result.message;
                    errorDiv.classList.remove('hidden');
                }
            } catch (error) {
                toggleLoading(false);
                errorDiv.innerText = "Error del servidor: Ocurrió un problema de conexión CORS o red.";
                errorDiv.classList.remove('hidden');
                console.error(error);
            }
        }

        // --- FUNCIONES ORIGINALES (Copiar) ---
        function copyToClipboard(elementId, btnElement) {
            const textToCopy = document.getElementById(elementId).innerText;
            const textArea = document.createElement("textarea");
            textArea.value = textToCopy;
            
            textArea.style.top = "0";
            textArea.style.left = "0";
            textArea.style.position = "fixed";

            document.body.appendChild(textArea);
            textArea.focus();
            textArea.select();

            try {
                const successful = document.execCommand('copy');
                if(successful) {
                    const originalHTML = btnElement.innerHTML;
                    if (btnElement.classList.contains('px-3')) {
                        btnElement.innerHTML = '<svg class="w-4 h-4 mr-1.5" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M5 13l4 4L19 7"></path></svg> ¡Copiado!';
                        btnElement.classList.add('text-green-600', 'border-green-600', 'bg-green-50');
                        btnElement.classList.remove('text-gray-600', 'border-gray-200', 'bg-gray-50');
                    } else {
                        btnElement.classList.add('text-green-500');
                    }

                    showToast();

                    setTimeout(() => {
                        btnElement.innerHTML = originalHTML;
                        if (btnElement.classList.contains('px-3')) {
                            btnElement.classList.remove('text-green-600', 'border-green-600', 'bg-green-50');
                            btnElement.classList.add('text-gray-600', 'border-gray-200', 'bg-gray-50');
                        } else {
                            btnElement.classList.remove('text-green-500');
                        }
                    }, 2000);
                }
            } catch (err) {
                console.error('No se pudo copiar el texto', err);
                alert("Error al copiar al portapapeles.");
            }
            document.body.removeChild(textArea);
        }

        function showToast() {
            const toast = document.getElementById("toast");
            toast.className = "toast show flex items-center justify-center space-x-2";
            setTimeout(function(){ 
                toast.className = toast.className.replace("show", ""); 
            }, 3000);
        }
